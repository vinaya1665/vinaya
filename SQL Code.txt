/*Table1*/
select COUNTRYREGION,ROUND((TOTALCASES/POPULATION),4) AS PERCENTAGE FROM (
select COUNTRYREGION ,POPULATION,TOTALCASES from WORLDOMETER2 WHERE POPULATION IS NOT NULL)
WHERE ROWNUM <= 5
ORDER BY PERCENTAGE DESC;

/*Table 2*/
select COUNTRYREGION,ROUND((ACTIVECASES/POPULATION),4) AS PERCENTAGE FROM (
select COUNTRYREGION ,POPULATION,ACTIVECASES from WORLDOMETER2 WHERE POPULATION IS NOT NULL)
WHERE ROWNUM <= 5
ORDER BY PERCENTAGE DESC;

/*Table 3*/
SELECT WHOREGION, SUM(TOTALDEATHS) AS NUMOFDEATHS 
FROM WORLDOMETER2
where rownum<=5
GROUP BY WHOREGION
ORDER BY NUMOFDEATHS DESC;
